// 审核状态流转

## 🗄️ 数据库扩展

### 新增表结构

```sql
-- 审核工作流表
CREATE TABLE article_review (
  id INT AUTO_INCREMENT PRIMARY KEY,
  article_id INT NOT NULL,
  review_level ENUM('initial', 'final', 'approve') DEFAULT 'initial',
  reviewer_id INT,
  status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
  comment TEXT,
  review_time DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 发布调度表
CREATE TABLE publish_schedule (
  id INT AUTO_INCREMENT PRIMARY KEY,
  article_id INT NOT NULL,
  platform ENUM('wechat', 'zhihu', 'xiaohongshu', 'toutiao') NOT NULL,
  scheduled_time DATETIME NOT NULL,
  status ENUM('scheduled', 'published', 'failed', 'cancelled') DEFAULT 'scheduled',
  published_at DATETIME,
  result_message TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 素材库表
CREATE TABLE media_library (
  id INT AUTO_INCREMENT PRIMARY KEY,
  file_name VARCHAR(255) NOT NULL,
  file_type ENUM('image', 'video', 'document') NOT NULL,
  file_path VARCHAR(500),
  file_url VARCHAR(500),
  file_size INT,
  tags JSON,
  uploader_id INT,
  usage_count INT DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 团队用户表
CREATE TABLE team_users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(50) NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  role ENUM('admin', 'editor', 'reviewer', 'viewer') DEFAULT 'viewer',
  email VARCHAR(100),
  last_login DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 操作日志表
CREATE TABLE operation_logs (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  action VARCHAR(50) NOT NULL,
  target_type VARCHAR(50),
  target_id INT,
  details JSON,
  ip_address VARCHAR(50),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```
